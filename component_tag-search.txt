# 頂層標籤

[[tags]]
name = "scp"
description = "項目檔案專用的標籤，需要被使用在在包括001提案，搞笑（-J），已歸檔（-ARC）、已解明（-EX）、以及ZH分部的原創（SCP-ZH-XXX）SCP項目上使用。"

[[tags]]
name = "故事"
description = "任何以SCP世界觀為背景撰寫的故事，但SCP項目檔案、GoI格式、網站頁面或補充資料並不是故事。"

[[tags]]
name= "goi格式"
description = "任何屬於關注組織（GOI）的創作的標籤，該頁面是從特定關注組織的角度編寫，並符合該組織規定的格式。goi格式的標籤也應該與該組織的標籤一同使用。"

[[tags]]
name= "附加資料"
description = "附加資料被定義為需要有另一份作品的上下文才能理解的頁面。這個頁面應該將自己的來源頁面設置為父頁面以保持文章的關聯性並建立頁面路徑，附加資料的設置只有在該頁面是由來源頁面的原作者親自或授權建立才會設置，致敬或模仿的作品應該被標記為故事，就算他們是以日誌或記錄的格式呈現。"

[[tags]]
name= "站點"
description = "站點頁面是紀錄基金會特定設施的中心頁面標籤，站點頁面提供儲存內部的項目檔案、站點歷史、在職人員等資訊。站點標籤應該與中心標籤一起使用。"

[[tags]]
name= "指導"
description = "是指導性質的頁面，多數是寫作指南以及有用的網站提示，或其他幫助投稿用戶的頁面。指導標籤只能在工作人員的批准下才能標註上去。請對照參閱「心得」標籤。"

[[tags]]
name= "心得"
description = "心得是根據作者個人的觀點，在某個特定的主題上撰寫的資訊。這些資源由專精各種不同方面(寫作，編碼，藝術作品等等)的作者所分享。心得的標籤可以由作者向工作人員確認後自行添加。"

[[tags]]
name= "新聞"
description = "新聞頁面包含了廣泛意義上所有關於基金會的新聞，從粉絲的新作品到最近發布的成功文章以及介於兩者之間的所有內容。也可能用在工作人員的正式通告上（只要它們未在主頁面或者討論帖發布完成）。"

[[tags]]
name= "資源"
description = "資源頁面列出了 SCP 世界各個方面的設定，能自由添加相關設定。"

[[tags]]
name= "作者頁面"
description = "這個標籤用在作者的個人頁面上。每名符合資格的創作者只能有一個作者頁面。"

[[tags]]
name= "藝術作品"
description = "藝術作品是關於基金會成員所創作的藝術作品合集。注意！創作的內容必須與基金會世界觀的事物有所關連，標記有「藝術作品」的頁面可以被網站成員投票評分。"

[[tags]]
name= "組件"
description = "透過include語法導入其他頁面，以添加新設計、樣式或功能的頁面。"

[[tags]]
name= "組件後端"
description = "被添加在「內部模板」頁面之上，這種頁面只會被導入至其他組件或組件後端的頁面裡面，但不會被作者直接使用。"

[[tags]]
name= "外觀主題"
description = "外觀主題是修改頁面外觀和感覺的CSS模板。"

[[tags]]
name= "段落"
description = "段落是導入到特定作品的頁面，頁面中只包含該作品內容。段落需要將該作品設置為 父頁面 。並且將網址放置於fragment: 類別中。段落頁的留存將與父頁面一致，如果該頁面是遺留頁或頁面的父頁面被刪除，這個頁面就將被刪除。"

[[tags]]
name= "討論區"
description = "討論區頁面是 Wikidot 論壇系統內部運行所需的頁面。它們只能由工作人員創建。"

[[tags]]
name= "遮蔽頁"
description = "遮蔽頁是用來為文件表現出一種需要授權的效果所添加的頁面。"

[[tags]]
name= "中心"
description = "中心頁面有著許多通往其他關連頁面的鏈接。所有SCP系列，基金會故事，以及一些頁面也都是中心頁。因為它是一個頂層標籤，所以中心頁標籤通常不會再增加其他的頂層標籤（例如 scp 或 故事）但競賽頁面除外。故事中心應該使用 _故事中心的隱藏標籤進行標記。"

[[tags]]
name= "工作台"
description = "工作台頁面是管理員或負責的版主們建立來通過維基模塊來管理特定頁面的頁面。雖然你可以在你的作者頁自由的使用這些模塊，但除非你是管理員或負責的版主，或者得他們的明確許可，請不要自己創建一個工作台。"

[[tags]]
name= "沙盒"
description = "此標籤目前只留作工作人員進行站點測試與實驗等用途。"

[[tags]]
name= "詩歌"

[[tags]]
name= "已歸檔"
description = "被取代、淘汰編號或因為某些因素被歸檔得頁面將會標註為已歸檔並且移往archived:類別中，加以鎖定並且禁止編輯。更多關於已歸檔頁面的相關資訊請見已歸檔SCP。"

[[tags]]
name= "已解明"
description = "已解明是一種特殊的「scp」副分類，並且帶有-EX後輟。應該要解明前最後一次分類的和「標準分級」或「機密分級」標籤一起使用。"

[[tags]]
name= "已退役"
description = "已退役代表此頁面已經在已解明徹底從SCP宇宙抹去存在，更多關於已退役頁面的相關資訊請見已退役SCP。"

[[tags]]
name= "001提案"
description = "擁有「001提案」的標籤文章都是可能的SCP-001項目。這個標籤需要跟其他的分級標籤一起使用，或在有必要時使用機密分級標籤。"

[[tags]]
name= "creepypasta"
description = "creepypasta-是與基金會無關的「故事」頁面，內容是講述原創的都市傳說或深夜怪談等等。creepypasta頁面必須附有「故事」標籤，特別注意本站收錄的「creepypasta」必須是由用戶自己撰寫而成的故事；如果想要分享在別處看到的「creepypasta」請至聊天室或閒聊版區。"

[[tags]]
name= "合作/協作"
description = "合作頁面是開放給所有網站成員添加內容的頁面，但必須注意，原作者與負責的工作人員有權修改或刪除其中不恰當內容的權利。"

[[tags]]
name= "合著/共同創作"
description = "合著的頁面是由兩位或更多作者合作的結果，除非另有說明否則不開放給其他網站成員增添內容。"

[[tags]]
name= "競賽"
description = "應用於網站內的競賽活動上。「競賽」標籤應該與「中心」一起建立在條目列表中，而不是蓋在其他相關的頁面上。這些競賽頁面只能在工作人員的批准下建立。"

[[tags]]
name= "必讀"
description = "標註在加入網站的成員必須詳細閱讀的指導頁面中。"

[[tags]]
name= "重新導向"
description = "重新導向是標註於將用戶重新定位到不同頁面的標籤，除了成人頁面以外這個標籤只由工作人員授權使用。"

[[tags]]
name= "_故事中心"
description = "標註在故事作品所組成的系列頁面中，需要與中心的標籤一起使用。"

[[tags]]
name= "實驗"
description = "用於「附加資料」頁面中主要由實驗日誌構成的頁面，就如SCP-914的實驗記錄。這標籤不應該直接用在scp頁面中，即使裡面有實驗記錄也一樣。需要與附加資料的標籤一起使用。"

[[tags]]
name= "探索"
description = "用於「附加資料」頁面中主要由對SCP的探索報告構成的頁面。這標籤不應該直接用在「scp」頁面上，即使裡面有探索報告也一樣。需要與附加資料的標籤一起使用。"

[[tags]]
name= "事故"
description = "用於「附加資料」頁面中主要由事故/事件報告構成的頁面，就如SCP- 555的事故記錄。這標籤不應該直接用在「scp」頁面上，即使裡面有事故報告也一樣。需要與附加資料的標籤一起使用。"

[[tags]]
name= "採訪"
description = "用於「附加資料」頁面中主要由對項目/人員進行的訪談記錄構成的頁面，就如SCP-277的採訪記錄277-A。這標籤不應該直接用在「scp」頁面上，即使裡面有訪談記錄也一樣。需要與附加資料的標籤一起使用。"
