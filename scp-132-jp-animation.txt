        var flag = a.c < a.l;
        if (flag)
          for (var i = 0; i < things.length; i++)
            if (things[i] != a.b && things[i].t == BOID && vLength2(vSub(a.p, things[i].p)) < 300) { flag = false; break; }
        if (!flag) a.g.removeChild(a.d);
        return flag;
      case CHAR:
        if (a.l <= a.c) a.g.removeChild(a.d);
        return a.c < a.l;
    }
    return true;
  }

  function render(a) {
    switch (a.t) {
      case BUBBLE:
        a.d.style.left = a.p.x * 0.1 + "rem";
        a.d.style.top = a.p.y * 0.1 + "rem";
        a.d.style.opacity = a.c < 5 ? 0 : Math.min(1, (a.l - a.c) / 10);
        return;
      case CHAR:
        a.d.style.left = a.p.x * 0.1 + "rem";
        a.d.style.top = a.p.y * 0.1 + "rem";
        a.d.style.opacity = Math.min(1, (a.l - a.c) / 10, a.c / 10);
        return;
      case BOID:
        a.d.style.left = a.p.x * 0.1 + "rem";
        a.d.style.top = a.p.y * 0.1 + "rem";
        a.d.style.opacity = 1 - Math.pow(1 - (Math.abs(Math.atan2(a.v.x, a.v.z)) / Math.PI), 4);

        a.d.style.transform = "rotateY(" + (-Math.atan2(a.v.x, a.v.z) / Math.PI * 180 + 90) + "deg)" + "rotateZ(" + (-Math.atan2(Math.sqrt(a.v.x * a.v.x + a.v.z * a.v.z), a.v.y) / Math.PI * 180 + 90) + "deg)";
        return;
    }
  }

  function float(a) {
    vAddTo(a.a, Vector(0, -FLOAT, 0));

    vAddTo(a.a, vScale(Vector(
      simplexNoise(a.p.x / 200, a.p.y / 200, t / 500),
      simplexNoise(a.p.x / 200, a.p.y / 200, (50 + t) / 500),

    vAddTo(a.a, vScale(vNormal(a.v), KEEP));

  function even(a) {
    vAddTo(a.a, Vector(0, a.v.y * -EVEN_Y, a.p.z * (Math.abs(a.v.z) + 0.001) * -EVEN_Z));

  function separation(a, things) {
    var visibleBoids = things.filter(function (x) {
      return a !== x && (x.t == BOID || x.t == CHAR) && vLength2(vSub(a.p, x.p)) < SIGHT * SIGHT;
    });
    for (var i = 0; i < visibleBoids.length; i++)
      vAddTo(a.a, vScale(vNormal(vSub(a.p, visibleBoids[i].p)), SEPARATION));
  }
  function alignment(a, visibleBoids) {
    if (visibleBoids.length !== 0) {
      var sumV = Vector(0, 0, 0);
      for (var i = 0; i < visibleBoids.length; i++) vAddTo(sumV, visibleBoids[i].v);
      var aveV = vScale(sumV, 1 / visibleBoids.length);
      vAddTo(a.a, vScale(vSub(aveV, a.v), ALIGNMENT));
    }
  }
  function cohesion(a, visibleBoids) {
    if (visibleBoids.length !== 0) {
      var sumP = Vector(0, 0, 0);
      for (var i = 0; i < visibleBoids.length; i++) vAddTo(sumP, visibleBoids[i].p);
      var aveP = vScale(sumP, 1 / visibleBoids.length);
      vAddTo(a.a, vScale(vSub(aveP, a.p), COHETION));
    }
  }
  function turn(a) {
    var wall = Vector(0, 0, 0);
    if (a.p.x < SIGHT)
      vAddTo(wall, Vector(1, 0, 0));
    if (WIDTH - SIGHT < a.p.x)
      vAddTo(wall, Vector(-1, 0, 0));
    if (a.p.y < SIGHT)
      vAddTo(wall, Vector(0, 0.2, 0));
    if (HEIGHT - SIGHT < a.p.y)
      vAddTo(wall, Vector(0, -0.2, 0));
    vAddTo(a.a, vScale(vCross(a.v, vCross(wall, a.v)), TURN));
  }
  function disturb(a, dist) {
    if (dist.c < 0 && dist.y1 < a.p.y && a.p.y < dist.y2 && dist.d * a.v.x < -0.025) {

  function apply(g) {
    var things = [];
    var p = Vector(
      Math.random() * WIDTH,
      Math.random() * HEIGHT,
      0
    );
    var v = Vector(
      (Math.random() - 0.5) * 0.125,
      (Math.random() - 0.5) * 0.015,
      (Math.random() - 0.5) * 0.05);

    for (var i = 0; i < 20; i++) {
      if (Math.random() < 0.5) {
        p = vAdd(p, Vector(
          (Math.random() - 0.5) * 40,
          (Math.random() - 0.5) * 40,
          0));
        v = vAdd(v, Vector(
          (Math.random() - 0.5) * 0.01,
          (Math.random() - 0.5) * 0.01,
          (Math.random() - 0.5) * 0.01));
        if (p.x < 0) p.x = -p.x;
        if (WIDTH < p.x) p.x = 2 * WIDTH - p.x;
        if (p.y < 0) p.y = -p.y;
        if (HEIGHT < p.y) p.y = 2 * HEIGHT - p.y;
      }
      else {
        p = Vector(
          Math.random() * WIDTH,
          Math.random() * HEIGHT,
          0);
        v = Vector(
          (Math.random() - 0.5) * 0.125,
          (Math.random() - 0.5) * 0.015,
          (Math.random() - 0.5) * 0.05);
      }
      things.push(Boid(p, v, g));
    }
    var last = performance.now();
    var dist = {
      c: 0,
      y1: 0,
      y2: 0,
      d: 1,
    };
    tick();
    function tick() {
      requestAnimationFrame(tick);
      var now = performance.now();
      var d = Math.min(now - last, 40);

      dist.c++;
      if (100 < dist.c) {
        var ncount = 0, pcount = 0;
        var y1 = Math.random() * (HEIGHT - 80) + 40;
        var y2 = y1 + Math.random() * 10 + 30;
        y1 = 2 * y1 - y2;
        for (var i = 0; i < things.length; i++) if (things[i].t == BOID) {
          if (things[i].v.x < 0) ncount++;
          if (0 < things[i].v.x) pcount++;
        }
        if (ncount < (ncount + pcount) * 0.2) dist = {
          c: -50,
          y1: y1,
          y2: y2,
          d: -1,
        };
        else if (pcount < (ncount + pcount) * 0.2) dist = {
          c: -50,
          y1: y1,
          y2: y2,
          d: 1,
        };
        else dist.c = 0;
      }

      for (var i = 0; i < things.length; i++) control(things[i], things, dist);
      var newThings = [];
      for (i = 0; i < things.length; i++) if (check(things[i], things)) newThings.push(things[i]);
      things = newThings;
      for (i = 0; i < things.length; i++) update(things[i], d);
      for (i = 0; i < things.length; i++) render(things[i]);

      last = now;
    }
  }

  var gs = document.getElementsByClassName("g");
  for (var j = 0; j < gs.length; j++)
    apply(gs[j]);
})();
